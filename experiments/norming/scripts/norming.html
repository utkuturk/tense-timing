Norming Analysis Report

1 Experiment Overview

This report analyzes the current norming dataset (results_feb_19.csv) for the sentence-picture fit task.

Participants saw a picture and rated how well a sentence fit the picture on a 1-7 scale.

Main design properties:

  • Event descriptions in past vs future tense.
  • Two verb classes in analysis: Regular and Irregular.
  • Timed response window: 15 seconds per trial.
  • Break screens inserted every 20 trials.
  • Filler trials used for data quality checks.

read_pcibex <- function(file,
                        encoding = "UTF-8",
                        exclude_controller_name = FALSE,
                        type_opts = list()) {
  con <- file(file, encoding = encoding)
  on.exit(close(con), add = TRUE)

  lines <- readLines(con, warn = FALSE)
  lines <- gsub(",+$", "", lines)
  lines <- lines[nzchar(lines)]

  parse_colnames <- function(x) {
    x <- x[grepl("^#\\s*\\d+\\.", x)]
    if (!length(x)) {
      return(character(0))
    }
    ids <- as.integer(sub("^#\\s*(\\d+)\\..*$", "\\1", x))
    vals <- sub("^#\\s*\\d+\\.\\s*", "", x)
    vals <- sub("\\.$", "", vals)
    out <- rep(NA_character_, max(ids))
    out[ids] <- vals
    out
  }

  header_blocks <- list()
  data_blocks <- list()
  i <- 1
  n <- length(lines)

  while (i <= n) {
    if (!grepl("^#", lines[i])) {
      i <- i + 1
      next
    }

    hdr <- character(0)
    while (i <= n && grepl("^#", lines[i])) {
      hdr <- c(hdr, lines[i])
      i <- i + 1
    }

    dat <- character(0)
    while (i <= n && !grepl("^#", lines[i])) {
      if (nzchar(lines[i])) {
        dat <- c(dat, lines[i])
      }
      i <- i + 1
    }

    if (length(dat)) {
      header_blocks[[length(header_blocks) + 1]] <- hdr
      data_blocks[[length(data_blocks) + 1]] <- dat
    }
  }

  if (!length(data_blocks)) {
    stop("No PCIbex data blocks found in file.")
  }

  ref_cols <- parse_colnames(header_blocks[[1]])
  block_colnames <- vector("list", length(data_blocks))
  block_data <- vector("list", length(data_blocks))

  for (j in seq_along(data_blocks)) {
    hdr_cols <- parse_colnames(header_blocks[[j]])
    max_len <- max(length(ref_cols), length(hdr_cols))
    cols <- rep(NA_character_, max_len)
    cols[seq_along(ref_cols)] <- ref_cols
    idx <- which(!is.na(hdr_cols))
    cols[idx] <- hdr_cols[idx]

    df <- utils::read.csv(
      text = paste(data_blocks[[j]], collapse = "\n"),
      header = FALSE,
      stringsAsFactors = FALSE,
      quote = "\"",
      fill = TRUE,
      comment.char = "",
      check.names = FALSE
    )

    if (length(cols) < ncol(df)) {
      cols <- c(cols, paste0("V", (length(cols) + 1):ncol(df)))
    } else if (length(cols) > ncol(df)) {
      cols <- cols[seq_len(ncol(df))]
    }

    cols[is.na(cols) | cols == ""] <- paste0("V", which(is.na(cols) | cols == ""))
    cols <- make.names(
      gsub("\\s+", "_", gsub("[^\\w\\s]", "", cols, perl = TRUE)),
      unique = TRUE
    )
    colnames(df) <- cols

    if (exclude_controller_name && "Controller_name" %in% colnames(df)) {
      df <- df[, setdiff(colnames(df), "Controller_name"), drop = FALSE]
    }

    block_colnames[[j]] <- colnames(df)
    block_data[[j]] <- df
  }

  all_cols <- unique(unlist(block_colnames, use.names = FALSE))
  for (j in seq_along(block_data)) {
    missing <- setdiff(all_cols, colnames(block_data[[j]]))
    if (length(missing)) {
      block_data[[j]][missing] <- NA
    }
    block_data[[j]] <- block_data[[j]][all_cols]
  }

  do.call(rbind, block_data)
}

clean_gender <- function(x) {
  lx <- tolower(trimws(as.character(x)))
  dplyr::case_when(
    grepl("^f", lx) | grepl("woman", lx) ~ "Female",
    grepl("^m", lx) | grepl("man", lx) ~ "Male",
    TRUE ~ "Other/Unclear"
  )
}

clean_native_language <- function(x) {
  lx <- tolower(trimws(as.character(x)))
  dplyr::if_else(grepl("english", lx), "English", "Other")
}

summarise_posterior <- function(x, label) {
  tibble::tibble(
    contrast = label,
    mean = mean(x),
    q2.5 = as.numeric(stats::quantile(x, 0.025)),
    q97.5 = as.numeric(stats::quantile(x, 0.975)),
    p_gt0 = mean(x > 0)
  )
}

7 Bayesian Modeling

The models below use maximal random-effects structure by participant and item:

  • Ordinal model: cumulative logit for rating_ord.
  • RT model: Gaussian model on log_rt.
artifact_path <- repo_path("analysis_output", "norming_model_artifacts.rds")
ord_model_path <- repo_path("analysis_output", "ord_model_tmp.rds")
rt_model_path <- repo_path("analysis_output", "rt_model_tmp.rds")

rstan_available <- requireNamespace("rstan", quietly = TRUE)

if (!rstan_available && file.exists(artifact_path)) {
  model_artifacts <- readRDS(artifact_path)
  ord_fixef <- model_artifacts$ord_fixef
  rt_fixef <- model_artifacts$rt_fixef
  pred_grid <- model_artifacts$pred_grid
  ord_pred_summary <- model_artifacts$ord_pred_summary
  rt_pred_summary <- model_artifacts$rt_pred_summary
  ord_expected_draws <- model_artifacts$ord_expected_draws
  rt_ms_draws <- model_artifacts$rt_ms_draws
  ord_diff_future_minus_past <- model_artifacts$ord_diff_future_minus_past
  rt_diff_future_minus_past <- model_artifacts$rt_diff_future_minus_past
  rt_ratio_future_over_past <- model_artifacts$rt_ratio_future_over_past
  posterior_contrasts <- model_artifacts$posterior_contrasts
  model_source <- paste(
    model_artifacts$model_source,
    "Using precomputed artifacts because rstan is unavailable in this render environment."
  )
} else if (!rstan_available && !file.exists(artifact_path)) {
  stop(
    paste0(
      "rstan is unavailable in this render environment and no precomputed artifacts were found at ",
      artifact_path,
      "."
    )
  )
} else {
  if (file.exists(ord_model_path) && file.exists(rt_model_path)) {
    ord_model <- readRDS(ord_model_path)
    rt_model <- readRDS(rt_model_path)
    model_source <- "Loaded existing models from analysis_output."
  } else {
    detected_cores <- parallel::detectCores(logical = FALSE)
    if (is.na(detected_cores) || detected_cores < 1) {
      detected_cores <- 2L
    }
    cores_to_use <- min(4L, as.integer(detected_cores))
    options(mc.cores = cores_to_use)

    backend_to_use <- if (requireNamespace("cmdstanr", quietly = TRUE)) "cmdstanr" else "rstan"

    ord_model <- brms::brm(
      rating_ord ~ tense * type + (1 + tense * type | id) + (1 + tense * type | picture),
      data = analysis_df,
      family = cumulative("logit"),
      backend = backend_to_use,
      chains = 4,
      cores = cores_to_use,
      iter = 2000,
      warmup = 1000,
      seed = 1001,
      control = list(adapt_delta = 0.95, max_treedepth = 12),
      refresh = 0
    )

    rt_model <- brms::brm(
      log_rt ~ tense * type + (1 + tense * type | id) + (1 + tense * type | picture),
      data = analysis_rt,
      family = gaussian(),
      backend = backend_to_use,
      chains = 4,
      cores = cores_to_use,
      iter = 2000,
      warmup = 1000,
      seed = 1002,
      control = list(adapt_delta = 0.95, max_treedepth = 12),
      refresh = 0
    )

    saveRDS(ord_model, ord_model_path)
    saveRDS(rt_model, rt_model_path)
    model_source <- "Models were fit in this render and saved to analysis_output."
  }

  ord_fixef <- as.data.frame(fixef(ord_model))
  ord_fixef <- cbind(term = rownames(ord_fixef), ord_fixef)
  rownames(ord_fixef) <- NULL

  rt_fixef <- as.data.frame(fixef(rt_model))
  rt_fixef <- cbind(term = rownames(rt_fixef), rt_fixef)
  rownames(rt_fixef) <- NULL

  pred_grid <- expand.grid(
    tense = levels(analysis_df$tense),
    type = levels(analysis_df$type),
    stringsAsFactors = FALSE
  ) %>%
    dplyr::mutate(
      tense = factor(tense, levels = levels(analysis_df$tense)),
      type = factor(type, levels = levels(analysis_df$type))
    )

  ord_epred <- posterior_epred(ord_model, newdata = pred_grid, re_formula = NA)
  rating_support <- seq_len(dim(ord_epred)[3])

  ord_expected_draws <- matrix(
    NA_real_,
    nrow = dim(ord_epred)[1],
    ncol = nrow(pred_grid)
  )
  for (i in seq_len(nrow(pred_grid))) {
    ord_expected_draws[, i] <- ord_epred[, i, ] %*% rating_support
  }

  ord_pred_summary <- pred_grid %>%
    dplyr::mutate(
      mean = apply(ord_expected_draws, 2, mean),
      lower = apply(ord_expected_draws, 2, stats::quantile, probs = 0.025),
      upper = apply(ord_expected_draws, 2, stats::quantile, probs = 0.975)
    )

  rt_linpred <- posterior_linpred(rt_model, newdata = pred_grid, re_formula = NA)
  rt_ms_draws <- exp(rt_linpred)
  rt_pred_summary <- pred_grid %>%
    dplyr::mutate(
      mean = apply(rt_ms_draws, 2, mean),
      lower = apply(rt_ms_draws, 2, stats::quantile, probs = 0.025),
      upper = apply(rt_ms_draws, 2, stats::quantile, probs = 0.975)
    )

  past_idx <- which(pred_grid$tense == "past")
  future_idx <- which(pred_grid$tense == "future")

  ord_past_draw <- rowMeans(ord_expected_draws[, past_idx, drop = FALSE])
  ord_future_draw <- rowMeans(ord_expected_draws[, future_idx, drop = FALSE])
  ord_diff_future_minus_past <- ord_future_draw - ord_past_draw

  rt_past_draw <- rowMeans(rt_ms_draws[, past_idx, drop = FALSE])
  rt_future_draw <- rowMeans(rt_ms_draws[, future_idx, drop = FALSE])
  rt_diff_future_minus_past <- rt_future_draw - rt_past_draw
  rt_ratio_future_over_past <- rt_future_draw / rt_past_draw

  posterior_contrasts <- dplyr::bind_rows(
    summarise_posterior(
      ord_diff_future_minus_past,
      "Expected rating difference: future - past"
    ),
    summarise_posterior(
      rt_diff_future_minus_past,
      "RT difference (ms): future - past"
    ),
    summarise_posterior(
      log(rt_ratio_future_over_past),
      "Log RT ratio: log(future/past)"
    )
  )

  model_artifacts <- list(
    ord_fixef = ord_fixef,
    rt_fixef = rt_fixef,
    pred_grid = pred_grid,
    ord_pred_summary = ord_pred_summary,
    rt_pred_summary = rt_pred_summary,
    ord_expected_draws = ord_expected_draws,
    rt_ms_draws = rt_ms_draws,
    ord_diff_future_minus_past = ord_diff_future_minus_past,
    rt_diff_future_minus_past = rt_diff_future_minus_past,
    rt_ratio_future_over_past = rt_ratio_future_over_past,
    posterior_contrasts = posterior_contrasts,
    model_source = model_source
  )
  saveRDS(model_artifacts, artifact_path)
}

In the RT model, the posterior mean future-minus-past difference is 13.7 ms, with a 95% credible interval of [-166.2, 191.3] ms.

The posterior probability that future RT is greater than past RT is 0.56, and the posterior RT ratio (future/past) is 1.007 with 95% CrI [0.933, 1.083].

This supports your main conclusion: there is no reliable RT difference between future and past, consistent with the idea that past and future are similarly accessible from the pictures.
